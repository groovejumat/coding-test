def solution(answers):
    #테스트용 유저를 생성
    first = [1, 2, 3, 4, 5]
    second = [2, 1, 2, 3, 2, 4, 2, 5, 2, 1, 2, 3, 2, 4, 2, 5]
    third = [3, 3, 1, 1, 2, 2, 4, 4, 5, 5, 3, 3, 1, 1, 2, 2, 4, 4, 5, 5]
    
    winner = [0,0,0]
    
    test = 0
    #first
    getans1=0
    for i in answers :
        #print(first[test])
        if first[test] == i :
            getans1=getans1+1
            #print("정답!")

        test = test+1
        
    print(getans1)
    
    #정답값을 넣어줍니다.
    winner[0]=getans1
        
    test = 0
    #second
    getans2=0
    for i in answers :
        #print(first[test])
        if second[test] == i :
            getans2=getans2+1
            #print("정답!")
            
        test = test+1
    
    winner[1]=getans2
    
    test = 0
    #third
    getans3=0
    for i in answers :
        #print(first[test])
        if third[test] == i :
            getans3=getans3+1
            #print("정답!")

            
        test = test+1
    
    #print(getans3)
    winner[2]=getans3
    
    
    print("위너의 정보 : ", winner)
    
    answer = []
    #위너의 정보를 통해서 하나의 배열을 만듭니다
    largest = winner[0]
    man = 1
    for i in winner:
         if i >= largest:
            answer.append(man)
            man = man+1
    
    return answer
